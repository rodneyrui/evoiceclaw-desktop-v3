二、如何构建“代码审核Skill”（实操指南）
2.1 设计Skill的基本功能
python
# skill_code_reviewer.py (概念示例)
class CodeReviewerSkill:
    """代码审核Skill，集成到CI/CD流程中"""
    
    def __init__(self, llm_router, git_client, pr_client):
        self.llm = llm_router
        self.git = git_client
        self.pr = pr_client
    
    async def review_pull_request(self, pr_number):
        # 1. 获取PR的变更文件列表
        files = await self.git.get_pr_files(pr_number)
        
        # 2. 对每个文件进行逐行审核
        for file in files:
            diff = file.patch  # 获取变更diff
            if self._should_review(file):
                comments = await self._review_file(file.filename, diff)
                # 3. 在PR中发布评论
                for comment in comments:
                    await self.pr.create_comment(
                        pr_number, 
                        file.filename, 
                        comment.line, 
                        comment.body
                    )
        
        # 4. 可选：添加总结评论
        summary = await self._generate_summary(pr_number)
        await self.pr.create_comment(pr_number, summary)
    
    async def _review_file(self, filename, diff):
        prompt = f"""你是一个资深的代码审核专家。请审查以下代码变更，重点关注：
1. 安全漏洞（SQL注入、XSS、路径遍历等）
2. 逻辑错误（空指针、边界条件、并发问题）
3. 代码规范（命名、注释、复杂度）
4. 性能隐患（不必要的循环、资源泄漏）

文件名：{filename}
变更内容（diff格式）：
{diff}

请以JSON数组格式输出审核意见，每个意见包含：
- line: 行号
- severity: "error" / "warning" / "suggestion"
- message: 具体问题描述
- suggestion: 修改建议（可选）

如果无问题，返回空数组。
"""
        result = await self.llm.complete(prompt, model="high_reasoning")
        return self._parse_comments(result)

2.4 确保审核质量：人工复核与反馈循环
